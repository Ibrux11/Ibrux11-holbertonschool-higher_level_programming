import unittest
from io import StringIO
import sys

# Supposons que MyList est défini dans un fichier appelé my_list.py
from my_list import MyList

class TestMyList(unittest.TestCase):
    
    def test_print_sorted(self):
        # Capture l'impression
        captured_output = StringIO()
        sys.stdout = captured_output
        
        my_list = MyList([3, 1, 2])
        my_list.print_sorted()
        
        # Rétablir stdout
        sys.stdout = sys.__stdout__
        
        # Vérifier si l'affichage est correct
        self.assertEqual(captured_output.getvalue().strip(), "[1, 2, 3]")
    
    def test_print_sorted_empty_list(self):
        # Capture l'impression
        captured_output = StringIO()
        sys.stdout = captured_output
        
        my_list = MyList([])
        my_list.print_sorted()
        
        # Rétablir stdout
        sys.stdout = sys.__stdout__
        
        # Vérifier si l'affichage est correct
        self.assertEqual(captured_output.getvalue().strip(), "[]")

    def test_print_sorted_with_negative_numbers(self):
        # Capture l'impression
        captured_output = StringIO()
        sys.stdout = captured_output
        
        my_list = MyList([3, -1, 2])
        my_list.print_sorted()
        
        # Rétablir stdout
        sys.stdout = sys.__stdout__
        
        # Vérifier si l'affichage est correct
        self.assertEqual(captured_output.getvalue().strip(), "[-1, 2, 3]")

if __name__ == '__main__':
    unittest.main()
